查找元素

  • myElement.text("Hello China!"); -- 设置 HTML 元素的内部文本 （js中，myElement.textContent = "Hello China!";）
  • var myText = myElement.text(); -- 获取 HTML 元素的内部文本 （js中，var myText = myElement.textContent || myElement.innerText;）
  • var myElement.html("

    Hello World

    "); -- 设置元素的 HTML 内容 （js中，var myElement.innerHTML = "

    Hello World

    ";）
  • var content = myElement.html(); -- 获取元素的 HTML 内容 （js中，var content = myElement.innerHTML;）
